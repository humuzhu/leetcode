func divide(a int, b int) int {
    if a == math.MinInt32 && b == -1 {
        return math.MaxInt32
    }

    sign := 0

    if a < 0 {
        sign++
    }

    if b < 0 {
        sign++
    }

    a, b = abs(a), abs(b)

    i := 0
    for a >= b {
        u, v := b, 1
        for a >= (u + u) {
            u += u
            v += v
        }
        a -= u // u = 2^j * b
        i += v // v = 2^j
    }

    if sign == 1 {
        return -i
    }
    return i
}

func abs(u int) int {
    if u > 0 {
        return u
    }
    return -u
}
